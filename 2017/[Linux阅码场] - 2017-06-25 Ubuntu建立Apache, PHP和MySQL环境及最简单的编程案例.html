下面我们在/var/www/html下面创建一个test-mysql.php，它会创建一个数据库和表：

<html>

<body>

<h1>mysql test</h1>

 

<?php

echo "creating database andtable";

 

$con =mysqli_connect("localhost","root","12345678");

if (mysqli_connect_errno($con))

  {

 die('Could not connect: ' . mysql_error());

  }

 

// Create database

mysqli_query($con, "CREATE DATABASEmy_db");

 

// Create table in my_db database

mysqli_select_db($con, "my_db");

$sql = "CREATE TABLE Persons

(

FirstName varchar(15),

LastName varchar(15),

Age int

)";

mysqli_query($con, $sql);

 

mysqli_close($con);

 

echo "<br>created database andtable";

?>


</body>

</html>
